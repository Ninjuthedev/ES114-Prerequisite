# Problem 22
def normalize(x):
    mylist = []
    minimumval = sorted(x)[0]
    maximumval = sorted(x)[-1]
    for i in x:
        normalizedval = ((i-minimumval)/(maximumval - minimumval))
        mylist.append(normalizedval)
    return mylist
n = eval(input())
print(f"Original Data: {n}")
print(f"Normalized data: {normalize(n)}")
plt.plot(n,normalize(n))
plt.xlabel("Original Data")
plt.ylabel("Normalized Data")
plt.title("Line Plot of Normalized Data")
plt.show()
